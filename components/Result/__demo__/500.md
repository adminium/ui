## en-US

```js
import { Result, Button } from '@adminium/arco-design';

const App = () => {
  return (
    <div>
      <Result
        status="500"
        subTitle="This page isn’t working."
        extra={<Button type="primary">Back</Button>}
      ></Result>
    </div>
  );
};

export default App;
```
